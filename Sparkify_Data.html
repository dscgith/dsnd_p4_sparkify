Sparkify Data Workspace¶

For reasons of clarity the Sparkify project has been split into two notebooks, this is part 1.
Data exploration and development of features will take place in this notebook, while modeling to predict customer churn is part 2, in the Sparkify_Modeling.ipynb notebook. For that reason the features engineered in this notebook are written to a Parquet file mini_sparkify_feature_data.parquet at the end of the notebook for easier data handling and further use in part 2.

The data used for this project is the mini_sparkify_event_data.json dataset provided by Udacity, containing 128MB of the full dataset available (12GB). Usage of a bigger dataset is not advised within the Udacity workspace for computation time reasons.

def transform_data(dataframe):
    """
    Transform, create and drop features from dataframe
    INPUTS:
    dataframe: Spark dataframe to transform
    """
    
    dataframe = dataframe.filter("userId != ''")
    
    # transform device information
    device_udf = udf(lambda x: "iPhone" if "iPhone" in x else 
                     ("Mac OS" if "Mac OS" in x else 
                      ("iPad" if "iPad" in x else 
                       ("Ubuntu" if "Ubuntu" in x else 
                        ("Windows" if "Windows" in x else "unknown")))))
    dataframe = dataframe.withColumn("device", device_udf(dataframe.userAgent))

    # transform gender information
    gender_udf = udf(lambda x: 1 if x == "F" else 0)
    dataframe = dataframe.withColumn("gender", gender_udf(dataframe.gender))

    # transform level information
    level_udf = udf(lambda x: 1 if x == "paid" else 0)
    dataframe = dataframe.withColumn("level", level_udf(dataframe.level))

    # transform states
    state_udf = udf(lambda x: re.findall(re.compile(r",\s(.+)"), x)[0])
    dataframe = dataframe.withColumn("location", state_udf(dataframe.location))

    # reduce dataframe to important features
    dataframe = dataframe.select('gender', 'length', 'level', "location", 'page',
                                 'sessionId', "device", 'ts', 'userId', "registration")

    # udfs for timestamp transformation
    get_timestamp = udf(lambda x: datetime.datetime.fromtimestamp(x / 1000.0).strftime("%Y-%m-%d %H:%M:%S"))
    get_timestamp_short = udf(lambda x: datetime.datetime.fromtimestamp(x / 1000.0).strftime("%Y-%m-%d"))
    get_hour_udf = udf(lambda x: int(datetime.datetime.fromtimestamp(x / 1000.0).hour))
    get_wday_udf = udf(lambda x: int(datetime.datetime.fromtimestamp(x / 1000.0).weekday()))
    get_day_udf = udf(lambda x: int(datetime.datetime.fromtimestamp(x / 1000.0).day))
    get_month_udf = udf(lambda x: int(datetime.datetime.fromtimestamp(x / 1000.0).month))
    get_year_udf = udf(lambda x: int(datetime.datetime.fromtimestamp(x / 1000.0).year))
    get_month_year_udf = udf(lambda x: str(datetime.datetime.fromtimestamp(x / 1000.0).year) 
                             + "/" + str(datetime.datetime.fromtimestamp(x / 1000.0).month))
    get_week_udf = udf(lambda x: str(datetime.date.isocalendar(datetime.datetime.fromtimestamp(x / 1000.0))[1]))
    get_week_year_udf = udf(lambda x: str(datetime.datetime.fromtimestamp(x / 1000.0).year) 
                            + "/" + str(datetime.date.isocalendar(datetime.datetime.fromtimestamp(x / 1000.0))[1]))
    get_timedelta_udf = udf(lambda x, y: (datetime.datetime.fromtimestamp(y / 1000.0) 
                                          - datetime.datetime.fromtimestamp(x / 1000.0)).total_seconds() 
                                          / 60 / 60 / 24 if x != 0 else 0)
    # get month/year of registration
    dataframe = dataframe.withColumn("reg_month_year", get_month_year_udf(dataframe.registration))
    dataframe = dataframe.withColumn("reg_week_year", get_week_year_udf(dataframe.registration))#.drop("registration")
    # get time features for timestamp
    dataframe = dataframe.withColumn("ts_hour", get_hour_udf(dataframe.ts).cast("int"))
    dataframe = dataframe.withColumn("ts_day", get_day_udf(dataframe.ts).cast("int"))
    dataframe = dataframe.withColumn("ts_week", get_week_udf(dataframe.ts).cast("int"))
    dataframe = dataframe.withColumn("ts_month", get_month_udf(dataframe.ts).cast("int"))
    dataframe = dataframe.withColumn("ts_year", get_year_udf(dataframe.ts).cast("int"))
    dataframe = dataframe.withColumn("ts_wday", get_wday_udf(dataframe.ts).cast("int"))
    dataframe = dataframe.withColumn("ts_week_year", get_week_year_udf(dataframe.ts))#.drop("ts")
    dataframe = dataframe.withColumn("membership_days", get_timedelta_udf(dataframe.registration, dataframe.ts).cast("int"))
    # previous timestamps
    dataframe = dataframe.withColumn("ts_prev", F.lag(dataframe.ts, default=0)
                                     .over(Window.partitionBy("userId").orderBy("ts")).cast("long"))
    # timedelta to previous interaction
    dataframe = dataframe.withColumn("ts_delta_minutes", F.round((get_timedelta_udf(dataframe.ts_prev, dataframe.ts) 
                                                                  * 24 * 60).cast("float"), 2))
    dataframe = dataframe.withColumn("ts_short", (get_timestamp_short(dataframe.ts)).cast("string"))
    dataframe = dataframe.withColumn("ts", (get_timestamp(dataframe.ts)).cast("string"))
    dataframe = dataframe.drop("ts_prev", "registration")

    # define churn based on cancellation confirmation event
    churn_udf = udf(lambda x: 1 if x == "Cancellation Confirmation" else 0)
    upgrade_udf = udf(lambda x: 1 if x == "Submit Upgrade" else 0)
    downgrade_udf = udf(lambda x: 1 if x == "Submit Downgrade" else 0)
    dataframe = dataframe.withColumn("downgrade", downgrade_udf("page").cast("int")) \
                         .withColumn("usr_down", F.sum("downgrade").over(Window.partitionBy("userId")).cast("int")) \
                         .withColumn("churn", churn_udf("page").cast("int")) \
                         .withColumn("usr_churn", F.sum("churn").over(Window.partitionBy("userId")).cast("int")) \
                         .withColumn("upgrade", upgrade_udf("page").cast("int")) \
                         .withColumn("usr_up", F.sum("upgrade").over(Window.partitionBy("userId")).cast("int")) 
    return dataframe

def plot_hist_agg_compare(pd_df, column, bins):

def plot_compare_churn(data, grpby, scale):
    """
    Plot a stacked bar chart for categorical data
    INPUTS:
    data: pre-filtered data with userId, usr_churn and category to plot
    grpby: category to plot
    scale: maximum y-axis value
    """
   
    locs = data.select(grpby).distinct()
    
    churn = locs.join(data.filter("usr_churn == 1").groupby(grpby) \
                      .agg(F.countDistinct("userId").alias("cnt")) \
                      .drop("userID"), on=grpby, how="left").fillna(0).toPandas()
    no_churn = locs.join(data.filter("usr_churn == 0").groupby(grpby) \
                         .agg(F.countDistinct("userId").alias("cnt")) \
                         .drop("userID"), on=grpby, how="left").fillna(0).toPandas()

    churn.cnt = churn.cnt.apply(lambda x: x / churn.cnt.sum() * 100)
    no_churn.cnt = no_churn.cnt.apply(lambda x: x / no_churn.cnt.sum() * 100)
    
    width = .4
    pos = np.arange(np.max([churn.shape[0], no_churn.shape[0]]))
    rotation = 0
    pwidth = 16
    if len(pos) > 10:
        rotation = 60
    plt.figure(figsize=(pwidth, 3));
    
    plt.bar(pos - (width/2), churn.cnt, color=[1, 0, 0], alpha=0.5, width=width, edgecolor='r');
    plt.bar(pos + (width/2), no_churn.cnt, color=[0, 0, 1], alpha=0.5, width=width, edgecolor='b');
    plt.title(grpby + " distribution of users");
    plt.ylabel("% of users");
    plt.ylim([0, scale]);
    plt.xticks(pos, churn[grpby].values, rotation=rotation);
    plt.legend(["churn", "stay"]);

Feature Engineering¶

During data exploration important features have been identified and preprocessed to be combined to a dataset to use with machine learning algorithms.
The process of creating different features was therefore included in Data exploration, Part 2 - comparing churning vs. staying users & feature creation. In this section those features are combined to the feature dataset.

The following features will be included, based on the userId:¶

  • membership_duration: length of membership in days
  • days_active: number of active days during membership
  • active_rate: percentage of active days during membership
  • paid_sub_share: percentage of membership with paid subscription (week based)
  • paid_use_share: percentage of interactions with paid subscription (interaction based)
  • page_*: total number of interactions/website visits
  • device_*: device used by the user for streaming
  • avg_session_len: average interactions in one session
  • avg_day_len: average daily listining time
  • gender: differentiate male/female users

Numeric encoding:¶

Columns with categorical values have been numerically encoded:

  • gender: male = 0 / female = 1
  • level: free = 0 / paid = 1

The features page* and device* have been pivoted to columns with count values.
